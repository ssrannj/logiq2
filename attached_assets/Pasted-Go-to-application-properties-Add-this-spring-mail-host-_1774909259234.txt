Create a Spring Boot EmailService using JavaMailSender.

Features:
- sendSimpleEmail(to, subject, body)
- sendPaymentSuccessEmail(to, orderId)
- sendWishlistNotificationEmail(to, productName)

Use jeyanthansaran@gmail.com as sender.

@Service
public class EmailService {

    @Autowired
    private JavaMailSender mailSender;

    private final String FROM_EMAIL = "jeyanthansaran@gmail.com";

    public void sendSimpleEmail(String to, String subject, String body) {
        SimpleMailMessage message = new SimpleMailMessage();
        message.setFrom(FROM_EMAIL);
        message.setTo(to);
        message.setSubject(subject);
        message.setText(body);
        mailSender.send(message);
    }

    public void sendPaymentSuccessEmail(String to, String orderId) {
        String subject = "Payment Verified - Thank You!";
        String body = "Your payment for Order ID " + orderId + " has been successfully verified.\n\nThank you for shopping with us!";
        sendSimpleEmail(to, subject, body);
    }

    public void sendWishlistNotificationEmail(String to, String productName) {
        String subject = "Item Back in Stock!";
        String body = productName + " is now back in stock. Grab it before it runs out!";
        sendSimpleEmail(to, subject, body);
    }
}
